function renderResults(data) { const list = document.getElementById('questionsList'); list.innerHTML = ''; // More robust error handling if (!data) { showError('No data received from server'); return; } // Handle different possible response structures let questions = data.questions; // If questions is an object with numeric keys, convert to array if (questions && typeof questions === 'object' && !Array.isArray(questions)) { questions = Object.values(questions); } // If questions is missing, try to find it elsewhere if (!questions || !Array.isArray(questions)) { if (data.predictions) questions = data.predictions; else if (data.results) questions = data.results; else { showError('Server response missing questions array: ' + JSON.stringify(data).substring(0, 200)); return; } } if (questions.length === 0) { showError('No questions were generated. Please try again with more detailed materials.'); return; } // Render each question questions.forEach((q, idx) => { const confidence = q.confidence || q.confidence_score || 70; const questionText = q.question || q.text || q.content || `Question ${idx + 1}`; const topic = q.topic || q.subject || q.category || "Key concept"; const confClass = confidence >= 80 ? 'conf-high' : confidence >= 65 ? 'conf-mid' : 'conf-low'; list.innerHTML += `
${q.num || idx + 1}
${escapeHtml(questionText)}
${confidence}% likely · ${escapeHtml(topic)}
`; }); // Handle the answer const answer = data.answer || data.model_answer || data.sample_answer || "No model answer was generated. Try providing more detailed syllabus content."; document.getElementById('answerBox').innerHTML = escapeHtml(answer).replace(/\n/g, '
'); document.getElementById('results').style.display = 'block'; document.getElementById('results').scrollIntoView({ behavior: 'smooth', block: 'start' }); } // Helper function to prevent XSS function escapeHtml(str) { if (!str) return ''; return str .replace(/&/g, '&') .replace(//g, '>') .replace(/"/g, '"') .replace(/'/g, '''); }
